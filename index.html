<script>
class FohowParticles {
constructor() {
this.canvas = document.getElementById('canvas');
this.ctx = this.canvas.getContext('2d');
this.particles = [];
this.animationId = null;
this.isAnimating = true;
this.colorSchemes = [
{ primary: '#ff8a00', secondary: '#e52e71' }, //
- -
{ primary: '#00b4db', secondary: '#0083b0' }, //
-
{ primary: '#c2e59c', secondary: '#64b3f4' }, //
-
{ primary: '#f4c4f3', secondary: '#fc67fa' } //
-
];
this.currentColorScheme = 0;
this.setup();
this.createParticles();
this.animate();
this.setupEventListeners();
}
setup() {
this.resize();
window.addEventListener('resize', () => this.resize());
}
resize() {
this.canvas.width = window.innerWidth;
this.canvas.height = window.innerHeight;
}
createParticles(count = 50) {
for (let i = 0; i < count; i++) {
this.particles.push({
x: Math.random() * this.canvas.width,
y: Math.random() * this.canvas.height,
vx: (Math.random() - 0.5) * 2,
vy: (Math.random() - 0.5) * 2,
radius: Math.random() * 3 + 1,
connections: 0
});
}
}
animate() {
if (!this.isAnimating) return;
this.ctx.fillStyle = 'rgba(15, 12, 41, 0.1)';
this.ctx.fillRect(0, 0, this.canvas.width,
this.canvas.height);
const colors =
this.colorSchemes[this.currentColorScheme];
//
this.particles.forEach(particle => {
particle.x += particle.vx;
particle.y += particle.vy;
//
if (particle.x <= 0 || particle.x >=
this.canvas.width) particle.vx *= -1;
if (particle.y <= 0 || particle.y >=
this.canvas.height) particle.vy *= -1;
//
this.ctx.beginPath();
this.ctx.arc(particle.x, particle.y, particle.radius,
0, Math.PI * 2);
this.ctx.fillStyle = colors.primary;
this.ctx.fill();
});
// -
this.particles.forEach((a, i) => {
this.particles.slice(i + 1).forEach(b => {
const dx = a.x - b.x;
const dy = a.y - b.y;
const distance = Math.sqrt(dx * dx + dy * dy);
if (distance < 150) {
this.ctx.beginPath();
this.ctx.strokeStyle =
`rgba(${this.hexToRgb(colors.secondary)}, ${1 - distance / 150})`;
this.ctx.lineWidth = 0.5;
this.ctx.moveTo(a.x, a.y);
this.ctx.lineTo(b.x, b.y);
this.ctx.stroke();
}
});
});
this.animationId = requestAnimationFrame(() =>
this.animate());
}
hexToRgb(hex) {
const result = /^#?([a-f\d]{2})([a-f\d]{2})([a-
f\d]{2})$/i.exec(hex);
return result ?
`${parseInt(result[1], 16)}, ${parseInt(result[2],
16)}, ${parseInt(result[3], 16)}`
: '255, 255, 255';
}
changeColorScheme() {
this.currentColorScheme = (this.currentColorScheme + 1) %
this.colorSchemes.length;
}
toggleAnimation() {
this.isAnimating = !this.isAnimating;
if (this.isAnimating) {
this.animate();
} else {
cancelAnimationFrame(this.animationId);
}
}
addParticles() {
this.createParticles(10);
}
resetParticles() {
this.particles = [];
this.createParticles(50);
}
setupEventListeners() {
this.canvas.addEventListener('click', (e) => {
// -
for (let i = 0; i < 5; i++) {
this.particles.push({
x: e.clientX + (Math.random() - 0.5) * 100,
y: e.clientY + (Math.random() - 0.5) * 100,
vx: (Math.random() - 0.5) * 4,
vy: (Math.random() - 0.5) * 4,
radius: Math.random() * 3 + 1,
connections: 0
});
}
});
}
}
//
let particlesSystem;
window.addEventListener('load', () => {
particlesSystem = new FohowParticles();
});
//
function changeColorScheme() {
particlesSystem.changeColorScheme();
}
function toggleAnimation() {
particlesSystem.toggleAnimation();
}
function addParticles() {
particlesSystem.addParticles();
}
function resetParticles() {
particlesSystem.resetParticles();
}
